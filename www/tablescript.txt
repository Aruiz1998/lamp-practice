CREATE TABLE `user` (
  `uid` int,
  `fname` varchar(50),
  `lname` varchar(50),
  `user` varchar(50),
  `Field` varchar(50),
  `created` datetime,
  PRIMARY KEY (`uid`)
);

CREATE TABLE `contacts` (
  `pid` int,
  `uid` int,
  `fname` varchar(50),
  `lname` varchar(50),
  `email` varchar(50),
  `phone` varchar(50),
  `created` datetime,
  PRIMARY KEY (`pid`),
  FOREIGN KEY (`uid`) REFERENCES `user`(`uid`)
);
